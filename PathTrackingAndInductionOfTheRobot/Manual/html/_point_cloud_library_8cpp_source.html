289  int j = 0; //クラスタのカウント

295  for (std::vector<pcl::PointIndices>::const_iterator it = cluster_indices.begin(); it != cluster_indices.end(); ++it) //クラスタを1塊ごとに出力

298  inputPointCloud->points[*pit].r = colors[j % 6][0];
299  inputPointCloud->points[*pit].g = colors[j % 6][1];
300  inputPointCloud->points[*pit].b = colors[j % 6][2];
301  cloud_cluster->points.push_back(inputPointCloud->points[*pit]);
302  }
303 
304  //最大のクラスタを探す．元のmac_clusterより新しいクラスタの方が大きければ新しいクラスタをmax_clusterとする(c76)
305  if (max_cluster->size() < cloud_cluster->size()){
306  pcl::copyPointCloud(*cloud_cluster, *max_cluster);
307  }

309  //cloud_cluster->width = cloud_cluster->points.size();
