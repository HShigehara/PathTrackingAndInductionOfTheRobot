67 void EV3Control::getVelocityinSec(double time_ms)
68 {
69  double k = 1000 / time_ms; //1秒あたりの速度に変換
70  if (flag_velocity == false){ //1フレーム目の処理
71  before = ev3_6dof; //速度の計算に必要な座標を保存
72  flag_velocity = true; //フラグをtrueにする
73  }
74  else{ //2フレーム目以降の処理
75  current = ev3_6dof; //現在の値を格納
76  //1フレームあたりの速度を計算
77  velocity = sqrt(pow((current.x-before.x),2)+pow((current.y-before.y),2)+pow((current.z-before.z),2));
78  //1秒あたりの速度に変換
79  velocity = velocity * k;
80  before = current; //現在の情報を過去の情報に更新する
81  }
82  //cout << "Velocity => " << velocity << "[mm/sec]" << endl; //確認用
83  return;
84 }

89 void EV3Control::getAverageVelocityAndYaw()
90 {
91  //double k = 1000.0 / time_ms; //このkを速度とヨー角にかけることで，1秒あたりの処理の場合に変換する
92 
93  //最初の5フレームはデータを保存するだけ
94  if (flag_average == false){ //5フレームの間の処理
95  if (count_average == 0){ //1フレーム目の処理
96  current_average.velocity = velocity;
97  current_average.yaw = ev3_6dof.yaw;
98  before5_average = current_average;
99  count_average++;
100  }
101  else if (count_average == 1){ //2フレーム目の処理
102  current_average.velocity = (velocity + before5_average.velocity) / 2.0;
103  current_average.yaw = (ev3_6dof.yaw + before5_average.yaw) / 2.0;
104  before4_average = current_average;
105  count_average++;
106  }
107  else if (count_average == 2){ //3フレーム目の処理
108  current_average.velocity = (velocity + before4_average.velocity + before5_average.velocity) / 3.0;
109  current_average.yaw = (ev3_6dof.yaw + before4_average.yaw + before5_average.yaw) / 3.0;
110  before3_average = current_average;
111  count_average++;
112  }
113  else if (count_average == 3){ //4フレーム目の処理
114  current_average.velocity = (velocity + before3_average.velocity + before4_average.velocity + before5_average.velocity) / 4.0;
115  current_average.yaw = (ev3_6dof.yaw + before3_average.yaw + before4_average.yaw + before5_average.yaw) / 4.0;
116  before2_average = current_average;
117  count_average++;
118  }
119  else if (count_average == 4){ //5フレーム目の処理
120  current_average.velocity = (velocity + before2_average.velocity + before3_average.velocity + before4_average.velocity + before5_average.velocity) / 5.0;
121  current_average.yaw = (ev3_6dof.yaw + before2_average.yaw + before3_average.yaw + before4_average.yaw + before5_average.yaw) / 5.0;
122  before1_average = current_average;
123  flag_average = true;
124  }
125  }
126  else{ //6フレーム以降
127  //現在の速度の平均を計算する
128  current_average.velocity = (velocity + before1_average.velocity + before2_average.velocity + before3_average.velocity + before4_average.velocity + before5_average.velocity) / 6.0; //現在と過去5フレーム分の速度から現在の速度を計算
129  //現在のヨー角の平均を計算する
130  current_average.yaw = (ev3_6dof.yaw + before1_average.yaw + before2_average.yaw + before3_average.yaw + before4_average.yaw + before5_average.yaw) / 6.0;
131 
132  //現在の速度とヨー角の平均を1フレームずつずらす
133  before1_average = current_average;
134  before2_average = before1_average;
135  before3_average = before2_average;
136  before4_average = before3_average;
137  before5_average = before4_average;
138  }
139  cout << "5 Frame Average [Velocity, Yaw] => " << "[ " << round(current_average.velocity) << " , " << round(current_average.yaw) << " ]" << endl;
140  //cout << current_average.velocity << endl;
141  return;
142 }
